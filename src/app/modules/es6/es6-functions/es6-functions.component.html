function parseInt(s: string, radix?: number): number

将字符串转换为十进制整数
s：需要转换的字符串
radix?：传入字符串采用的进制数，范围2--36，不指定默认为10表示十进制。
    parseInt('100',2) = 4
    parseInt('100') = 100
    parseInt('100a') = 100
    parseInt(100a) error
    parseInt('a00') = NaN
    parseInt('') = NaN
    parseInt(null) = NaN

function parseFloat(s: string): number

将字符串转换为浮点数
s：需要转换的字符串
    parseFloat('100') = 100
    parseFloat('100.00') = 100
    parseFloat('100.001a') = 100.001
    parseFloat(100.001a) error
    parseFloat('a100.001') = NaN
    parseFloat('') = NaN
    parseFloat(null) = NaN

function isNaN(number: number): boolean

判断给定值是否是数值
number：待判断的数值
    isNaN('100') = false
    isNaN('100.00a') = true
    isNaN(100.00a) error
    isNaN('') = false
    isNaN(null) = false

function isFinite(number: number): boolean;

检查其参数是否是有限大。
number：待判断的参数
    isFinite('1.22')) = true
    isFinite('1.22a')) = false
    isFinite(1.22a) error
    isFinite(1/0) = false
    isFinite(1/3) = true

function encodeURI(uri: string): string;

把字符串作为URI进行编码。(不会对本身属于URI的特殊字符进行编码)
uri：需要编码的文本
    encodeURI("https://github.com/dodobelieve?prp='最靓的仔'") = "https://github.com/dodobelieve?prp='%E6%9C%80%E9%9D%93%E7%9A%84%E4%BB%94'"
    encodeURI("") = ""
    encodeURI(",/?:@&=+$#") = ",/?:@&=+$#"
    encodeURI("aa") = "aa"

function decodeURI(encodedURI: string): string;

对encodeURI()函数编码过的URI进行解码。
encodedURI：编码后的URI
    decodeURI("https://github.com/dodobelieve?prp='%E6%9C%80%E9%9D%93%E7%9A%84%E4%BB%94'") = "https://github.com/dodobelieve?prp='最靓的仔'"
    decodeURI("") = ""
    decodeURI("aa") = "aa"

function encodeURIComponent(uriComponent: string): string;

把字符串作为 URI 组件进行编码。(会对本身属于URI的特殊字符进行编码)
uriComponent：需要编码的文本
    encodeURIComponent("https://github.com/dodobelieve?prp='最靓的仔'") = "https%3A%2F%2Fgithub.com%2Fdodobelieve%3Fprp%3D'%E6%9C%80%E9%9D%93%E7%9A%84%E4%BB%94'"
    encodeURIComponent("") = ""
    encodeURIComponent(",/?:@&=+$#") = "%2C%2F%3F%3A%40%26%3D%2B%24%23"
    encodeURIComponent("aa") = "aa"

function decodeURIComponent(encodedURIComponent: string): string;

对encodeURIComponent()函数编码的URI进行解码。
encodedURIComponent：需要解码的文本
    decodeURIComponent("https%3A%2F%2Fgithub.com%2Fdodobelieve%3Fprp%3D'%E6%9C%80%E9%9D%93%E7%9A%84%E4%BB%94'") = "https://github.com/dodobelieve?prp='最靓的仔'"
    decodeURIComponent("https://github.com/dodobelieve?prp='%E6%9C%80%E9%9D%93%E7%9A%84%E4%BB%94'") = "https://github.com/dodobelieve?prp='最靓的仔'"
    decodeURIComponent("") = ""
    decodeURIComponent("%2C%2F%3F%3A%40%26%3D%2B%24%23") = ",/?:@&=+$#"
    decodeURIComponent("aa") = "aa"
对于通过encodeURI()编码的文本decodeURIComponent()也能正确解码： 
    decodeURIComponent("https://github.com/dodobelieve?prp='%E6%9C%80%E9%9D%93%E7%9A%84%E4%BB%94'") = "https://github.com/dodobelieve?prp='最靓的仔'"
对于通过encodeURIComponent()编码的文本decodeURI()不能完全解码(特殊字符部分)： 
    decodeURI("https%3A%2F%2Fgithub.com%2Fdodobelieve%3Fprp%3D'%E6%9C%80%E9%9D%93%E7%9A%84%E4%BB%94'") = "https%3A%2F%2Fgithub.com%2Fdodobelieve%3Fprp%3D'最靓的仔'"

function escape(string: string): string;

对字符串进行编码
string：需要编码的文本
(已不推荐使用，推荐使用encodeURI()或encodeURIComponent())
    escape("https://github.com/dodobelieve?prp='最靓的仔'") = "https%3A//github.com/dodobelieve%3Fprp%3D%27%u6700%u9753%u7684%u4ED4%27"
    escape("") = ""
    escape(",/?:@&=+$#") = "%2C/%3F%3A@%26%3D+%24%23"
    escape("aa") = "aa"

function unescape(string: string): string;

可对通过 escape() 编码的字符串进行解码。
string：需要解码的文本
(已不推荐使用，推荐使用decodeURI()或decodeURIComponent())

    unescape("") = ""
    unescape("%2C%2F%3F%3A%40%26%3D%2B%24%23") = ",/?:@&=+$#"
    unescape("aa") = "aa"
